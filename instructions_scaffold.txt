Project Scaffold Outline

Below is the directory structure with each file and a brief description of its intended contents. Use this as your guide when creating the actual files.

├── README.md
├── main.py # Entry point: loads cli_controller and starts MainMenu
├── app/
│
│ ├── core/ # Business logic layer (framework- and CLI-agnostic)
│ │ ├── init.py
│ │ ├── account.py # Account-related operations (usage/billing data)
│ │ ├── call.py # Dialing logic (initiate outbound calls)
│ │ ├── message.py # SMS sending logic
│ │ └── phone_number.py # Search & purchase logic:
│ │ • raw-HTTP search batcher
│ │ • dedupe and stop criteria
│ │ • purchase via Twilio SDK
│
│ ├── gateways/ # Adapters for external systems
│ │ ├── init.py
│ │ ├── config.py # Load .env or env vars, expose config object
│ │ ├── file_logger.py # Optional file-based logging (structured JSON)
│ │ └── twilio_gateway.py # Wrap Twilio SDK client creation and calls
│
│ ├── interfaces/ # CLI layer
│ │ ├── cli_controller.py # Bootstraps menus, runs the menu loop
│ │ └── menus/
│ │ ├── init.py
│ │ ├── base_menu.py # BaseMenu class: clear screen, render header, prompt utilities
│ │ ├── main_menu.py # MainMenu: choices 1–3, dispatch to Purchase, Manage, Settings
│ │
│ │ ├── purchase/ # Purchase Numbers nested flow
│ │ │ ├── purchase_menu.py # PurchaseMenu: inline prompts for country/type/mode/capabilities
│ │ │ ├── locality_input_menu.py # LocalityInputMenu: render table of regions/localities
│ │ │ ├── search_progress_menu.py # SearchProgressMenu: show Rich progress bar
│ │ │ ├── search_results_menu.py # SearchResultsMenu: paginated table, sort/save, select indexes
│ │ │ └── purchase_confirm_menu.py # PurchaseConfirmMenu: display chosen numbers, confirm purchase
│ │
│ │ ├── manage/ # Manage Numbers nested flow
│ │ │ ├── manage_menu.py # ManageMenu: render active numbers table
│ │ │ ├── number_actions_menu.py # NumberActionsMenu: list call/SMS/logs/config/release options
│ │ │ ├── call_menu.py # CallMenu: prompt for destination, display result
│ │ │ ├── sms_menu.py # SmsMenu: prompt for destination/message, result
│ │ │ ├── logs_menu.py # LogsMenu: choose messaging or call logs, paginated table
│ │ │ ├── config_menu.py # ConfigMenu: display current settings table
│ │ │ ├── voice_config_menu.py # VoiceConfigMenu: choose URL/TwiML/SIP trunk
│ │ │ ├── messaging_config_menu.py # MessagingConfigMenu: URL/TwiML/Service management
│ │ │ └── release_menu.py # ReleaseMenu: confirm and release number
│ │
│ │ ├── settings/ # Settings & Admin standard menu
│ │ │ ├── settings_menu.py # SettingsMenu: list 3.1–3.9, dispatch accordingly
│ │ │ ├── billing_menu.py # BillingMenu: display usage & cost summary
│ │ │ ├── security_menu.py # SecurityMenu: list/manage API keys & IP whitelist
│ │ │ ├── subaccount_menu.py # SubaccountMenu: table of subaccounts, switch/create
│ │ │ ├── devtools_menu.py # DevToolsMenu: webhook/sandbox/test-credentials info
│ │ │ ├── logs_menu.py # AdminLogsMenu: global logs table
│ │ │ ├── advanced_search_menu.py # AdvancedSearchMenu: deep number search table
│ │ │ ├── config_management_menu.py # ConfigMgmtMenu: bulk config templates/audit
│ │ │ └── diagnostics_menu.py # DiagnosticsMenu: rate-limit/webhook error tables
│
│ ├── models/ # Data models and lookup tables
│ │ ├── init.py
│ │ ├── account_model.py # Data classes for account info
│ │ ├── country_data.py # Unified COUNTRY_DATA dict: names, pricing, regions, area codes
│ │ ├── message_model.py # Data classes for SMS logs/records
│ │ └── phone_number_model.py # Data classes for number record & search session metadata
│
│ ├── services/ # High-level orchestration of use cases
│ │ ├── init.py
│ │ ├── account_service.py # Wrappers around core.account for CLI usage
│ │ ├── messaging_service.py # Wrappers around core.message
│ │ ├── number_service.py # Combines core.phone_number + twilio_gateway
│ │ └── voice_service.py # Wrappers around core.call
│
│ └── use_cases/ # Optional orchestrators that tie menus → services
│ ├── init.py
│ ├── purchase_flow.py # Logic that sequence core + services for PurchaseMenu
│ ├── manage_flow.py # Logic for manage_menu → number actions
│ └── settings_flow.py # Logic for settings_menu → submenus
